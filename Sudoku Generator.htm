import java.util.ArrayList; public class SudokuGenerator { public static void main(String[] args) { int[][] board = new int[9][9]; ArrayList base = new ArrayList(); for (int i = 1; i <= 9; i++) { base.add(i); } for (int row = 0; row < 9; row++) { int shift = (row * 3 + row / 3) % 9; for (int col = 0; col < 9; col++) { int index = (col + shift) % 9; board[row][col] = base.get(index); } } printBoard(board); } public static void printBoard(int[][] board) { for (int row = 0; row < 9; row++) { if (row % 3 == 0 && row != 0) { System.out.println("---------------------"); } for (int col = 0; col < 9; col++) { if (col % 3 == 0 && col != 0) { System.out.print("| "); } System.out.print(board[row][col] + " "); } System.out.println(); } } }
